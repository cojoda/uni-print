\makeatletter
\def\@maketitle{%
  \begin{center}%
    \let \footnote \thanks
    {\LARGE \@title \par}
    {\vskip .5em \large \labsection}
    {\vskip .5em \large \homework}
    {\vskip .5em \large \github}
    {\vskip .5em \large \@author}
    {\vskip .5em \large \@date}
  \end{center}
\par \vskip .5em}
